Palettes

map-get(map-get($palettes, purple), light);
@function palette($palette, $tone: 'base') {
  @return map-get(map-get($palettes, $palette), $tone);
}
h1, h2, h3 {
  font-family: $font-title;
  color: palette(purple);
}
h4, h5, h6 {
  color: palette(purple, light);
}
